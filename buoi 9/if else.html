> ( lớn hơn )
        < ( nhỏ hơn )
        == ( so sánh bằng tương đối ) chỉ so sánh giá trị
        = ( phép gán )
        === ( so sánh bằng tuyệt đối ) vừa so sánh giá trị và
        so sánh thêm cả kiểu dữ liệu
        
        if (điều kiện) {
            // nếu điều kiện trả về true thì thực hiện các câu lệnh
            // trong {} đầu tiên
        } else {
            // nếu điều kiện trả về true thì thực hiện các câu lệnh
            // trong {} sau từ khóa else
        }
        
        // if lồng nhau:
        if (điều kiện 1) {
            // nếu điều kiện 1 == true thì chạy các câu lệnh ở đây
        } else if (điều kiện 2) {
            // nếu điều kiện 2 == true thì chạy các câu lệnh ở đây
        } else if (điều kiện 3) {
            // nếu điều kiện 3 == true thì chạy các câu lệnh ở đây
        } ...
        else {
            // các trường hợp còn lại
        }

        *** chú ý
        nếu nhiều điều kiện cùng đúng thì if hay else if nào
        phía trước sẽ được chạy mà không chạy các else if phía sau
            and: &&
            or: ||
