        //
        khi mà margin-top bằng margin-bottom 
        đồng thời margin-left bằng margin-right 
        thì mình có cách viết rút gọn :
        margin: A B;
        trong đó A là giá trị margin-top ( margin-bottom )
        B là giá trị margin-left ( margin-right )

        //
        margin-top khác margin-bottom 
        đồng thời margin-left bằng margin-right
        margin: A B C;
        trong đó A là giá trị margin-top
        B là giá trị margin-left ( margin-right )
        C là giá trị margin-bottom

        //
        margin: auto;

        margin: 30px auto;
